# Router 4 Configuration file

telnet 192.168.0.1

conf t
interface lo
no ip address 127.0.0.1/8
ip address 10.0.6.24/32
exit
interface eth0.2
no shutdown
no ip address 192.168.2.1/24
ip address 10.0.6.1/30
exit
interface eth0.1
no shutdown
no ip address 192.168.1.1/24
ip address 10.0.0.6/24
exit
router ospf
router-id 4
area 6 range 10.0.6.0/24
network 10.0.6.0/30 area 10.0.6.0
exit
exit
show ip route
